CREATE OR REPLACE VIEW VW_DATOS_UNION AS( 

SELECT  EN.CODIGO AS CODIGO_ENTE,ENTE.NIT_ENTE,
  PERFIL.CODIGO CODIGO_PERFIL,soli.FECHA_MODIFICACION,soli.FECHA_MODIFICACION + 183 as "ACTUALIZA",
  
  repre.PRIMER_NOMBRE , repre.OTROS_NOMBRE,repre.PRIMER_APELLIDO,repre.SEGUNDO_APELLIDO
FROM VW_DATOS_GENERALES_ENTE ENTE,
  TB_PERFIL_ENTE PERFIL,
  TB_ENTE EN,
  TB_SOLICITUD_ENTE soli,TB_REPRESENTANTE_ENTE repre
WHERE  ENTE.CODIGO_SOLICITUD = PERFIL.CODIGO_SOLICITUD AND
soli.nit_ente= en.nit_ente AND
repre.codigo=PERFIL.CODIGO  and
en.nit_ente=ENTE.NIT_ENTE
AND PERFIL.CODIGO         =
  (SELECT MAX(CODIGO)
  FROM TB_PERFIL_ENTE
  WHERE CODIGO_SOLICITUD= ENTE.CODIGO_SOLICITUD
  ));
